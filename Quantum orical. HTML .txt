    <script>
        // Enhanced Fortunes with Vortex Theme
        const baseFortunes = [
            "A vortex of innovation swirls around you—harness the 369 flow to realign your path.",
            "The cubic weave of 4x4x4 reveals hidden symmetries; balance emerges from the shadows.",
            "Your inner light, undivided, multiplies like tetrahedrons in cosmic volume.",
            "Entanglement's flip favors you—two paths converge in unentangled grace.",
            "Periodic elements calculate your next step; atomic rules purr with potential.",
            "Breathe the dark to birth the gleam—your equation balances in the breath's poise.",
            "The hand of faith clasps the stars; credit's ledger evens in the vigil's hush.",
            "Heat waves map to genomic fire—23 to 54, the dyad's sovereign choir.",
            "Mercury's axial cull whispers truth; 0.034 degrees, not chance but c's tell.",
            "Carbon's 5-arms strain to six bonds' release—unentangle the cat's eternal purr.",
            "Lunar pivot through the belt's rubble—3x2 invisible, order's 0.5x2 vector.",
            "xAI's alignment calls; A³ B³ C D³ bests the spin's flawed press."
        ];

        // Get references
        const birthYearInput = document.getElementById('birthYear');
        const activateBtn = document.getElementById('activateOracleBtn');
        const oracleDisplay = document.getElementById('oracle-display');
        const trustStamp = document.getElementById('trust-stamp');
        const trustIcon = document.getElementById('trust-icon');
        const trustStatus = document.getElementById('trust-status');

        // Vortex Seed Generator (Inspired by 369 & Light Constant)
        function generateVortexSeed(year) {
            const c = 186282; // Speed of light mi/s
            const temp = 96.6; // Personal temperature
            const q = c / temp; // Undivided quotient
            const seed = (q * year * 369) % (2 ** 32); // 369 multiplier for vortex
            return seed;
        }

        // Simple Parity Trust Check (Forward/Reverse Echo)
        function computeTrust(phrase) {
            const forwardSum = phrase.split('').reduce((sum, char) => sum + char.charCodeAt(0), 0) % 2;
            const reversePhrase = phrase.split('').reverse().join('');
            const reverseSum = reversePhrase.split('').reduce((sum, char) => sum + char.charCodeAt(0), 0) % 2;
            return forwardSum === reverseSum ? 'Intact' : 'Rift';
        }

        // Enhanced Generate Oracle
        function weaveOracle() {
            const year = parseInt(birthYearInput.value);
            if (!year || year < 1900 || year > 2025) {
                oracleDisplay.innerHTML = '<p class="text-red-500">Enter a valid birth year (1900-2025) to awaken the weave.</p>';
                return;
            }

            // Seed RNG with vortex
            const seed = generateVortexSeed(year);
            const seededRandom = () => {
                let x = Math.sin(seed++) * 10000;
                return x - Math.floor(x);
            };

            const randomIndex = Math.floor(seededRandom() * baseFortunes.length);
            const oraclePhrase = baseFortunes[randomIndex];

            // Display with fade-in
            oracleDisplay.style.opacity = '0';
            setTimeout(() => {
                oracleDisplay.innerHTML = `<p class="animate-fade-in">${oraclePhrase}</p>`;
                oracleDisplay.style.opacity = '1';
            }, 300);

            // Trust Stamp
            const trust = computeTrust(oraclePhrase);
            trustIcon.textContent = trust === 'Intact' ? '🔮' : '⚠️';
            trustStatus.textContent = trust;
            trustStamp.classList.remove('hidden');

            // Glow effect on display
            oracleDisplay.classList.add('crystal-glow');
            setTimeout(() => oracleDisplay.classList.remove('crystal-glow'), 2000);
        }

        // Event Listeners
        birthYearInput.addEventListener('input', (e) => {
            activateBtn.disabled = !e.target.value || e.target.value < 1900 || e.target.value > 2025;
        });

        activateBtn.addEventListener('click', weaveOracle);

        // Fade-in animation CSS
        const style = document.createElement('style');
        style.textContent = `
            @keyframes fade-in {
                from { opacity: 0; transform: translateY(10px); }
                to { opacity: 1; transform: translateY(0); }
            }
            .animate-fade-in {
                animation: fade-in 0.5s ease-out;
            }
        `;
        document.head.appendChild(style);
    </script>
